#In order to compare Opioid Overdoses vs Median Household Income, I chose to use scatterplots with a treadline to show correlation and patters.
x = df_op['MedianHouseholdIncome']
y = df_op['FPOO-Rate']
fig,ax = plt.subplots()
ax.scatter(x, y, alpha=0.5, c = 'b')  
fig.suptitle('Opiod Overdoses Vs Median Household Income')   # Graph title
ax.xaxis.set_label_text('Median Household Income')  # x-axis caption
ax.yaxis.set_label_text('Opioid overdoses per 100,000 Residents')       # y-axis caption
fig.set_size_inches(7,5)
sns.lmplot(x='MedianHouseholdIncome', y='FPOO-Rate', data = df_op);
plt.show()
